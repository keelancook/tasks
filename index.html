<script>

    let cur = 0, sel = null, arcFocus = null;

    const STORAGE_KEY = 'analog_db';


    let db = {

        today: JSON.parse(localStorage.getItem(STORAGE_KEY+'_t')) || Array(10).fill().map(()=>({t:"", s:0})),

        next: JSON.parse(localStorage.getItem(STORAGE_KEY+'_n')) || Array(10).fill().map(()=>({t:"", s:0})),

        someday: JSON.parse(localStorage.getItem(STORAGE_KEY+'_s')) || Array(10).fill().map(()=>({t:"", s:0})),

        archive: JSON.parse(localStorage.getItem(STORAGE_KEY+'_a')) || [],

        lastActiveDate: localStorage.getItem(STORAGE_KEY+'_lastDate') || ""

    };


    function save() { 

        localStorage.setItem(STORAGE_KEY+'_t', JSON.stringify(db.today)); 

        localStorage.setItem(STORAGE_KEY+'_n', JSON.stringify(db.next)); 

        localStorage.setItem(STORAGE_KEY+'_s', JSON.stringify(db.someday)); 

        localStorage.setItem(STORAGE_KEY+'_a', JSON.stringify(db.archive)); 

        localStorage.setItem(STORAGE_KEY+'_lastDate', db.lastActiveDate);

    }


    function getShortDate(dObj) { return `${dObj.toLocaleDateString('en-US', { month: 'short' })} ${dObj.getDate()}, ${dObj.getFullYear()}`; }


    function render() {

        const views = ['today', 'next', 'someday', 'archive'];

        const v = views[cur];

        const el = document.getElementById('card');

        el.className = 'card ' + (arcFocus ? 'c-archive' : 'c-' + v);


        if(cur === 3 && !arcFocus) {

            const gear = `<svg class="settings-trigger" onclick="openSettings()" viewBox="0 0 24 24"><path d="M19.14 12.94c.04-.3.06-.61.06-.94 0-.32-.02-.64-.07-.94l2.03-1.58c.18-.14.23-.41.12-.61l-1.92-3.32c-.12-.22-.37-.29-.59-.22l-2.39.96c-.5-.38-1.03-.7-1.62-.94l-.36-2.54c-.04-.24-.24-.41-.48-.41h-3.84c-.24 0-.43.17-.47.41l-.36 2.54c-.59.24-1.13.57-1.62.94l-2.39-.96c-.22-.08-.47 0-.59.22L2.74 8.87c-.12.21-.08.47.12.61l2.03 1.58c-.05.3-.09.63-.09.94s.02.64.07.94l-2.03 1.58c-.18.14-.23.41-.12.61l1.92 3.32c.12.22.37.29.59.22l2.39-.96c.5.38 1.03.7 1.62.94l.36 2.54c.05.24.24.41.48.41h3.84c.24 0 .44-.17.47-.41l.36-2.54c.59-.24 1.13-.56 1.62-.94l2.39.96c.22.08.47 0 .59-.22l1.92-3.32c.12-.22.07-.47-.12-.61l-2.01-1.58zM12 15.6c-1.98 0-3.6-1.62-3.6-3.6s1.62-3.6 3.6-3.6 3.6 1.62 3.6 3.6-1.62 3.6-3.6 3.6z"/></svg>`;

            el.innerHTML = `<div class="hdr"><b>Archive</b>${gear}</div><div id="arch-list" style="overflow-y:auto; height:calc(100% - 100px)"></div>`;

            db.archive.slice().reverse().forEach((e, i) => {

                let realIdx = db.archive.length - 1 - i;

                let r = document.createElement('div'); r.className='arch-item';

                r.innerHTML = `<div style="font-size:28px" onclick="viewArc(${realIdx})">${e.d}</div><div style="color:red" onclick="delArch(event, ${realIdx})">×</div>`;

                document.getElementById('arch-list').appendChild(r);

            });

            return;

        }


        const data = arcFocus ? arcFocus.tasks : db[v];

        const title = arcFocus ? 'History' : v.charAt(0).toUpperCase() + v.slice(1);

        let sub = arcFocus ? arcFocus.d : (cur === 0 ? getShortDate(new Date()) : (cur === 1 ? "Priority" : "Ideas"));


        el.innerHTML = `<div class="hdr"><b>${title}</b><span>${sub}</span></div><ul id="list" class="${cur===0?'today-list':''}"></ul><div id="btn-container"></div>`;

        updateTodayButton();


        const list = document.getElementById('list');

        const arrowSVG = `<svg class="del-arrow" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3.5" stroke-linecap="round" stroke-linejoin="round"><line x1="7" y1="12" x2="17" y2="12"></line><polyline points="12 7 17 12 12 17"></polyline></svg>`;

        data.forEach((item, i) => {

            let li = document.createElement('li');

            let icon = item.s==1 ? '<div class="dot-in"></div>' : (item.s==2 ? arrowSVG : '');

            li.innerHTML = `<div class="btn-c" onclick="tgl(${i})">${icon}</div><input type="text" value="${item.t}" oninput="upd(${i},this.value)" class="${item.s==1?'done':''}">

                            <div class="ctrls"><span class="dots" onclick="openM(${i})">⋮</span></div>`;

            list.appendChild(li);

        });

    }


    function updateTodayButton() {

        const container = document.getElementById('btn-container');

        if(!container || cur !== 0 || arcFocus) return;

        const hasTasks = db.today.some(x => x.t.trim() !== "");

        container.innerHTML = `<div style="padding:10px; text-align:center;"><button class="fin-btn" onclick="${hasTasks?'fin()':'viewLastArc()'}">${hasTasks?'Finish Day':'Review Yesterday'}</button></div>`;

    }


    function upd(i,v) { 

        if(arcFocus) return; 

        db[['today','next','someday','archive'][cur]][i].t=v; 

        save(); 

        if(cur === 0) updateTodayButton(); 

    }


    function tgl(i) { if(arcFocus)return; let v=['today','next','someday','archive'][cur]; db[v][i].s=(db[v][i].s+1)%3; save(); render(); }

    function viewArc(i) { arcFocus = { tasks: db.archive[i].tasks, d: db.archive[i].d, idx: i }; render(); }

    function delArch(e, i) { e.stopPropagation(); if(confirm("Delete card?")) { db.archive.splice(i, 1); save(); render(); } }

    function viewLastArc() { if(db.archive.length > 0) viewArc(db.archive.length - 1); else alert("No history"); }


    function openM(i) { 

        sel=i; const container = document.getElementById('m-options-container'); container.innerHTML = '';

        const addOption = (label, action) => {

            const div = document.createElement('div'); div.className = 'm-item'; div.innerText = label; div.onclick = () => mAct(action);

            container.appendChild(div);

        };

        if(arcFocus) {

            addOption("Migrate to Today", "today"); addOption("Migrate to Next", "next");

        } else {

            const v = ['today','next','someday'][cur];

            if(v !== 'today') addOption("Move to Today", "today");

            if(v !== 'next') addOption("Move to Next", "next");

            if(v !== 'someday') addOption("Move to Someday", "someday");

            addOption("Delete Task", "del");

        }

        document.getElementById('menu').classList.add('open'); document.getElementById('ov').style.display='block'; 

    }


    function mAct(a) { 

        let src = arcFocus ? arcFocus.tasks : db[['today', 'next', 'someday'][cur]];

        if (a === 'del') { src[sel] = {t:"", s:0}; } 

        else {

            let ti = db[a].findIndex(x => x.t.trim() === "");

            let newTask = {t: src[sel].t, s: arcFocus ? 2 : src[sel].s};

            if(ti !== -1) db[a][ti] = newTask; else db[a].push(newTask);

            if(arcFocus) src[sel].s = 2; else src[sel] = {t:"", s:0};

        }

        save(); render(); closeAll();

    }


    function fin() { if(!confirm("Archive today?")) return; db.archive.push({d: getShortDate(new Date()), tasks: JSON.parse(JSON.stringify(db.today))}); db.today = Array(10).fill().map(()=>({t:"", s:0})); save(); render(); }

    function openSettings() { document.getElementById('settings-menu').classList.add('open'); document.getElementById('ov').style.display='block'; }

    function closeAll() { document.getElementById('menu').classList.remove('open'); document.getElementById('settings-menu').classList.remove('open'); document.getElementById('ov').style.display='none'; }

    function exportData() { const a = document.createElement('a'); a.href = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(db)); a.download = "analog_backup.json"; a.click(); }

    function importData() { document.getElementById('fileInput').click(); }

    document.getElementById('fileInput').onchange = e => { const reader = new FileReader(); reader.onload = ev => { db = JSON.parse(ev.target.result); save(); render(); closeAll(); }; reader.readAsText(e.target.files[0]); };


    let sX = 0;

    document.addEventListener('touchstart', e => sX = e.touches[0].clientX);

    document.addEventListener('touchend', e => {

        let diff = e.changedTouches[0].clientX - sX;

        if(Math.abs(diff) > 70) { if(arcFocus){ arcFocus=null; render(); return; } jump(diff > 0 ? -1 : 1); }

    });

    function jump(dir) { cur = (cur + dir + 4) % 4; render(); }


    render();

</script>
